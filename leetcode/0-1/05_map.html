给定两个数组，编写一个函数来计算它们的交集。

      示例 1：

      输入：nums1 = [1,2,2,1], nums2 = [2,2]
      输出：[2]

      var intersection = function(nums1, nums2) {
		    let m = new Map();
		    let res = []
		    nums1.forEach(n => {
	        m.set(n, true)
		    })
		    nums2.forEach(n => {
	        if(m.has(n)) {
	          res.push(n)
	          m.delete(n)
	        }
		    })
		    return res;
			};
